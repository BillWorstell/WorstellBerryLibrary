# From the repo root
function Add-TocImage {
  param(
    [Parameter(Mandatory=$true)][string]$Slug,              # e.g. 'JohnAdams_McCullough'
    [Parameter(Mandatory=$true)][string]$ShelfPath          # e.g. 'docs/books/BookCase01/Shelf01'
  )
  $src = "images/$Slug.jpg"
  $dst = Join-Path $ShelfPath "$Slug.jpg"
  if (!(Test-Path $src)) { throw "Missing $src" }
  Copy-Item $src $dst -Force
  git add $dst
  git commit -m "Add curated TOC image for $Slug"
  git push
}

# Example for the Ellis book:
 Add-TocImage -Slug 'AmericanSphinx_Ellis_TOC' -ShelfPath 'docs/books/BookCase01/Shelf01'
